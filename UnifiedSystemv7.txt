# ─────────────────────────────────────────────────────────────────────────────
# v6.1 — PERCEPTUAL DIFFERENCE HASH (dHash)
# ─────────────────────────────────────────────────────────────────────────────
def dhash(frame: np.ndarray, hash_size: int = 8) -> Optional[int]:
    """
    Difference hash — fast perceptual similarity fingerprint.
    Identical hash between pre- and post-action frames means the screen did
    NOT change: the action had zero visible effect.  Used as the closed-loop
    success/failure signal in ActionExecutor and the NavigatorAgent run loop.
    """
    if frame is None:
        return None
    try:
        gray    = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        resized = cv2.resize(gray, (hash_size + 1, hash_size))
        diff    = resized[:, 1:] > resized[:, :-1]
        return sum(2 ** i for i, v in enumerate(diff.flatten()) if v)
    except Exception:
        return None

class _JSONFormatter(logging.Formatter):
    def format(self, record: logging.LogRecord) -> str:
        return json.dumps({
            "ts":        datetime.now(timezone.utc).isoformat(),
            "level":     record.levelname,
            "component": getattr(record, "component", "core"),
            "msg":       record.getMessage(),
            "node":      os.environ.get("NAVIGATOR_NODE_ID", platform.node()),
        })


class _ANSIFormatter(logging.Formatter):
    _COLOURS = {
        "DEBUG":    "\033[90m",
        "INFO":     "\033[0m",
        "WARNING":  "\033[33m",
        "ERROR":    "\033[31m",
        "CRITICAL": "\033[35m",
    }
    _RESET = "\033[0m"

    def format(self, record: logging.LogRecord) -> str:
        colour    = self._COLOURS.get(record.levelname, "")
        ts        = datetime.now().strftime("%H:%M:%S.%f")[:-3]
        component = getattr(record, "component", "core").upper().ljust(14)
        return f"{colour}[{ts}] {component} {record.getMessage()}{self._RESET}"


def setup_logging(
    log_file: str = "navigator.log",
    level: int = logging.INFO,
) -> logging.Logger:
    logger = logging.getLogger("navigator")
    logger.setLevel(level)
    if logger.handlers:
        return logger

    ch = logging.StreamHandler(sys.stdout)
    ch.setFormatter(_ANSIFormatter())
    logger.addHandler(ch)

    fh = logging.handlers.RotatingFileHandler(
        log_file, maxBytes=20 * 1024 * 1024, backupCount=5, encoding="utf-8"
    )
    fh.setFormatter(_JSONFormatter())
    logger.addHandler(fh)
    return logger


_logger: logging.Logger = logging.getLogger("navigator")

def _resolve_base_type(annotation: Any) -> Optional[Type]:
    """
    Robustly extract the base scalar type from an annotation that may be a
    raw type, a string forward-reference, or an Optional[T] / Union[T, None].
    """
    if isinstance(annotation, str):
        try:
            annotation = eval(annotation, vars(typing))  # noqa: S307
        except Exception:
            return None

    origin = typing.get_origin(annotation)
    if origin is Union:
        args = [a for a in typing.get_args(annotation) if a is not type(None)]
        if args:
            annotation = args[0]
        else:
            return None

    if annotation in (int, float, bool, str):
        return annotation
    return None

        if not _TOML_AVAILABLE:
            raise ConfigurationError(
                "tomllib/tomli not available; cannot load TOML config"
            )
        with open(path, "rb") as f:
            data = _tomllib.load(f)
        valid = {f.name for f in dc_fields(cls)}
        return cls(**{k: v for k, v in data.items() if k in valid})

    @classmethod
    def from_env(cls) -> "Config":
        cfg = cls()
        try:
            hints = typing.get_type_hints(cls)
        except Exception:
            hints = {}

        for fld in dc_fields(cls):
            env_key = f"NAV_{fld.name.upper()}"
            val = os.environ.get(env_key)
            if val is None:
                continue

            annotation = hints.get(fld.name, fld.type)
            base_type  = _resolve_base_type(annotation)

            try:
                if base_type is bool:
                    setattr(cfg, fld.name, val.lower() in ("1", "true", "yes"))
                elif base_type is int:
                    setattr(cfg, fld.name, int(val))
                elif base_type is float:
                    setattr(cfg, fld.name, float(val))
                else:
                    setattr(cfg, fld.name, val)
            except (ValueError, TypeError):
                _log("config", f"Cannot coerce {env_key}={val!r}", "warning")

        return cfg

        valid_fields = {f.name for f in dc_fields(cls)}
        return cls(**{k: v for k, v in raw_copy.items() if k in valid_fields})

# CIRCUIT BREAKER  (v5 canonical: monotonic clock, typed enum, locked state)
# ─────────────────────────────────────────────────────────────────────────────
class CircuitBreaker:
    """Prevents cascading API failures with open/half-open/closed states."""

    class State(Enum):
        CLOSED    = auto()
        OPEN      = auto()
        HALF_OPEN = auto()

    def __init__(self, threshold: int = 5, reset_sec: int = 60) -> None:
        self.threshold  = threshold
        self.reset_sec  = reset_sec
        self._failures  = 0
        self._state     = self.State.CLOSED
        self._opened_at: Optional[float] = None
        self._lock      = threading.Lock()

    @property
    def state(self) -> "CircuitBreaker.State":
        with self._lock:
            if self._state == self.State.OPEN:
                elapsed = time.monotonic() - (self._opened_at or 0)
                if elapsed > self.reset_sec:
                    self._state = self.State.HALF_OPEN
                    _log("circuit", "Entering HALF_OPEN — probing API")
        return self._state

    def record_success(self) -> None:
        with self._lock:
            self._failures = 0
            self._state    = self.State.CLOSED

    def record_failure(self) -> None:
        with self._lock:
            self._failures += 1
            if self._failures >= self.threshold:
                self._state     = self.State.OPEN
                self._opened_at = time.monotonic()
                _log("circuit",
                     f"Circuit OPEN after {self._failures} failures", "error")

    @contextmanager
    def guard(self):
        if self.state == self.State.OPEN:
            raise CircuitOpenError("Circuit breaker is OPEN — API calls blocked")
        try:
            yield
            self.record_success()
        except (APIConnectionError, APIStatusError, RateLimitError):
            self.record_failure()
            raise


# ─────────────────────────────────────────────────────────────────────────────
# ADAPTIVE RATE LIMITER  (v5 canonical: throttle multiplier feedback loop)
# ─────────────────────────────────────────────────────────────────────────────
class AdaptiveRateLimiter:
    def __init__(self, calls_per_min: int) -> None:
        self._cpm          = calls_per_min
        self._tokens       = float(calls_per_min)
        self._last_refill  = time.monotonic()
        self._lock         = threading.Lock()
        self._throttle_mul = 1.0

    def _refill(self) -> None:
        now   = time.monotonic()
        delta = now - self._last_refill
        added = (delta / 60.0) * (self._cpm / self._throttle_mul)
        self._tokens      = min(float(self._cpm), self._tokens + added)
        self._last_refill = now

    def acquire(self, timeout: float = 30.0) -> None:
        deadline = time.monotonic() + timeout
        while True:
            with self._lock:
                self._refill()
                if self._tokens >= 1.0:
                    self._tokens -= 1.0
                    return
            wait = 0.5
            if time.monotonic() + wait > deadline:
                raise APIError("Rate limiter timeout — too many pending calls")
            time.sleep(wait)

    def on_rate_limited(self) -> None:
        with self._lock:
            self._throttle_mul = min(self._throttle_mul * 1.5, 8.0)
            _log("ratelimit",
                 f"Throttle multiplier → {self._throttle_mul:.1f}x", "warning")

    def on_success(self) -> None:
        with self._lock:
            self._throttle_mul = max(1.0, self._throttle_mul * 0.95)


# ─────────────────────────────────────────────────────────────────────────────
# CRYPTOGRAPHIC AUDIT LOG  (v5 canonical: HMAC embedded in entry)
# ─────────────────────────────────────────────────────────────────────────────
class AuditLog:
    """Append-only JSONL audit log with per-entry HMAC signatures."""

    def __init__(self, path: str, hmac_key: str, enabled: bool = True) -> None:
        self._path    = Path(path)
        self._key     = hmac_key
        self._enabled = enabled
        self._lock    = threading.Lock()

    def _sign(self, payload: str) -> str:
        return hmac.new(
            self._key.encode(), payload.encode(), hashlib.sha256
        ).hexdigest()

    def write(self, event_type: str, data: Dict[str, Any]) -> None:
        if not self._enabled:
            return
        entry: Dict[str, Any] = {
            "ts":         datetime.now(timezone.utc).isoformat(),
            "event_type": event_type,
            "node":       os.environ.get("NAVIGATOR_NODE_ID", platform.node()),
            "data":       data,
        }
        payload       = json.dumps(entry, default=str, sort_keys=True)
        entry["hmac"] = self._sign(payload)
        with self._lock:
            with self._path.open("a", encoding="utf-8") as f:
                f.write(json.dumps(entry, default=str) + "\n")

    def verify_integrity(self) -> Tuple[int, int]:
        """Returns (valid_count, invalid_count)."""
        valid = invalid = 0
        if not self._path.exists():
            return 0, 0
        with self._path.open("r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    entry    = json.loads(line)
                    sig      = entry.pop("hmac", "")
                    payload  = json.dumps(entry, default=str, sort_keys=True)
                    expected = self._sign(payload)
                    if hmac.compare_digest(sig, expected):
                        valid   += 1
                    else:
                        invalid += 1
                except Exception:
                    invalid += 1
        return valid, invalid


# ─────────────────────────────────────────────────────────────────────────────
# SQLITE CONNECTION POOL  (v5 canonical: thread-local WAL connections)
# ─────────────────────────────────────────────────────────────────────────────
class _SQLitePool:
    def __init__(self, db_path: str) -> None:
        self._db_path = db_path
        self._local   = threading.local()

    def get(self) -> sqlite3.Connection:
        conn = getattr(self._local, "conn", None)
        if conn is None:
            conn = sqlite3.connect(
                self._db_path,
                check_same_thread=False,
                isolation_level=None,
            )
            conn.row_factory = sqlite3.Row
            conn.execute("PRAGMA journal_mode=WAL")
            conn.execute("PRAGMA synchronous=NORMAL")
            conn.execute("PRAGMA foreign_keys=ON")
            self._local.conn = conn
        return conn

    @contextmanager
    def transaction(self):
        conn = self.get()
        conn.execute("BEGIN")
        try:
            yield conn
            conn.execute("COMMIT")
        except Exception:
            conn.execute("ROLLBACK")
            raise

# EMBEDDING PROVIDER  (v5 canonical: full fallback chain xAI→Voyage→Nomic→BOW)
# ─────────────────────────────────────────────────────────────────────────────
class EmbeddingProvider:
    """
    Multi-provider embedding backend with automatic fallback chain:
      xAI → Voyage → Nomic → BOW

    Each provider returns a fixed-dimension float32 numpy vector.
    BOW fallback is always available with no external API.
    """

    _PROVIDER_DEFAULTS: Dict[str, Tuple[str, int]] = {
        "xai":    ("text-embedding-3-small", 1536),
        "voyage": ("voyage-3-lite",          512),
        "nomic":  ("nomic-embed-text-v1",    768),
    }

    def __init__(self, cfg: Config) -> None:
        self._cfg      = cfg
        self._provider = cfg.embedding_provider.lower()
        self._dim      = cfg.embedding_dim
        self._lock     = threading.Lock()

        # Vocabulary for BOW fallback
        self._bow_vocab: Dict[str, int] = {}
        self._bow_lock  = threading.Lock()

        self._client: Optional[openai.OpenAI] = None
        self._embed_model: str = cfg.embedding_model

        self._init_client()

    def _init_client(self) -> None:
        p = self._provider
        if p == "xai":
            key = os.environ.get("XAI_API_KEY", "")
            if key:
                self._client = openai.OpenAI(
                    api_key=key,
                    base_url="https://api.x.ai/v1",
                    timeout=15.0,
                )
                _log("embed", f"Embedding provider: xAI ({self._embed_model})")
            else:
                _log("embed", "XAI_API_KEY missing — falling back to BOW", "warning")
                self._provider = "bow"

        elif p == "voyage":
            key = os.environ.get("VOYAGE_API_KEY", "")
            if key:
                self._client = openai.OpenAI(
                    api_key=key,
                    base_url="https://api.voyageai.com/v1",
                    timeout=15.0,
                )
                default_model, default_dim = self._PROVIDER_DEFAULTS["voyage"]
                if self._embed_model == "text-embedding-3-small":
                    self._embed_model = default_model
                self._dim = default_dim
                _log("embed", f"Embedding provider: Voyage ({self._embed_model})")
            else:
                _log("embed", "VOYAGE_API_KEY missing — falling back to xAI", "warning")
                self._provider = "xai"
                self._init_client()

        elif p == "nomic":
            key = os.environ.get("NOMIC_API_KEY", "")
            if key:
                self._client = openai.OpenAI(
                    api_key=key,
                    base_url="https://api-atlas.nomic.ai/v1",
                    timeout=15.0,
                )
                default_model, default_dim = self._PROVIDER_DEFAULTS["nomic"]
                if self._embed_model == "text-embedding-3-small":
                    self._embed_model = default_model
                self._dim = default_dim
                _log("embed", f"Embedding provider: Nomic ({self._embed_model})")
            else:
                _log("embed", "NOMIC_API_KEY missing — falling back to BOW", "warning")
                self._provider = "bow"

        else:
            self._provider = "bow"
            _log("embed", "Embedding provider: BOW (legacy fallback)")

    def embed(self, text: str) -> np.ndarray:
        """Return a normalised float32 embedding vector."""
        if self._provider == "bow" or self._client is None:
            return self._bow_embed(text)
        try:
            return self._api_embed(text)
        except Exception as exc:
            _log("embed", f"API embedding failed ({exc}), using BOW fallback", "warning")
            return self._bow_embed(text)

    @property
    def dim(self) -> int:
        return self._dim

    def _api_embed(self, text: str) -> np.ndarray:
        assert self._client is not None
        if self._provider == "nomic":
            text = f"search_document: {text}"
        resp = self._client.embeddings.create(
            model=self._embed_model,
            input=text[:8000],
        )
        vec  = np.array(resp.data[0].embedding, dtype=np.float32)
        norm = np.linalg.norm(vec)
        return vec / norm if norm > 0 else vec

    def _bow_embed(self, text: str) -> np.ndarray:
        """Bag-of-words fallback with stable persistent vocabulary."""
        tokens = re.findall(r"\b[a-zA-Z]{3,}\b", text.lower())
        with self._bow_lock:
            for t in tokens:
                if t not in self._bow_vocab:
                    self._bow_vocab[t] = len(self._bow_vocab)
            vocab_snapshot = dict(self._bow_vocab)

        dim = max(len(vocab_snapshot), 1)
        vec = np.zeros(dim, dtype=np.float32)
        for t in tokens:
            if t in vocab_snapshot:
                vec[vocab_snapshot[t]] += 1.0
        norm = np.linalg.norm(vec)
        return vec / norm if norm > 0 else vec

    @staticmethod
    def cosine(a: np.ndarray, b: np.ndarray) -> float:
        """Cosine similarity; zero-pads shorter vector (BOW compat)."""
        la, lb = len(a), len(b)
        if la == 0 or lb == 0:
            return 0.0
        if la < lb:
            a = np.concatenate([a, np.zeros(lb - la, dtype=np.float32)])
        elif lb < la:
            b = np.concatenate([b, np.zeros(la - lb, dtype=np.float32)])
        denom = np.linalg.norm(a) * np.linalg.norm(b)
        return float(np.dot(a, b) / denom) if denom > 0 else 0.0

# EMBEDDING CACHE  (v5 canonical: phash-keyed, thread-safe, stats tracking)
# ─────────────────────────────────────────────────────────────────────────────
class EmbeddingCache:
    """
    LRU cache mapping perceptual-hash hex-string → np.ndarray embedding.
    Avoids redundant API calls for duplicate/near-duplicate screen frames.
    Thread-safe.
    """

    def __init__(self, maxsize: int = 64) -> None:
        self._cache: OrderedDict[str, np.ndarray] = OrderedDict()
        self._maxsize = maxsize
        self._lock    = threading.Lock()
        self._hits    = 0
        self._misses  = 0

    def _key(self, phash: np.ndarray) -> str:
        return phash.tobytes().hex()

    def get(self, phash: np.ndarray) -> Optional[np.ndarray]:
        k = self._key(phash)
        with self._lock:
            if k in self._cache:
                self._cache.move_to_end(k)
                self._hits += 1
                return self._cache[k]
            self._misses += 1
            return None

    def put(self, phash: np.ndarray, embedding: np.ndarray) -> None:
        k = self._key(phash)
        with self._lock:
            if k in self._cache:
                self._cache.move_to_end(k)
            else:
                if len(self._cache) >= self._maxsize:
                    self._cache.popitem(last=False)
                self._cache[k] = embedding

    @property
    def stats(self) -> Dict[str, int]:
        return {"hits": self._hits, "misses": self._misses, "size": len(self._cache)}


# ─────────────────────────────────────────────────────────────────────────────
# LONG-TERM MEMORY  (v5 canonical: SQLite + Redis backends, real embeddings)
# ─────────────────────────────────────────────────────────────────────────────
@dataclass
class MemoryFact:
    text:       str
    embedding:  Optional[np.ndarray]
    created_at: float
    session_id: str
    importance: float = 1.0


class _MemoryBackend:
    def add(self, fact: MemoryFact) -> None: ...
    def get_all(self) -> List[MemoryFact]: ...
    def delete_expired(self, ttl: float) -> int: ...


class SQLiteMemoryBackend(_MemoryBackend):
    def __init__(self, db_path: str) -> None:
        self._pool = _SQLitePool(db_path)
        self._init_db()

    def _init_db(self) -> None:
        conn = self._pool.get()
        conn.executescript("""
            CREATE TABLE IF NOT EXISTS facts (
                id          TEXT PRIMARY KEY,
                text        TEXT NOT NULL,
                embedding   BLOB,
                created_at  REAL NOT NULL,
                session_id  TEXT NOT NULL,
                importance  REAL DEFAULT 1.0
            );
            CREATE INDEX IF NOT EXISTS idx_facts_created ON facts(created_at);

            CREATE TABLE IF NOT EXISTS vocab (
                token   TEXT PRIMARY KEY,
                idx     INTEGER NOT NULL UNIQUE
            );
        """)

    def add(self, fact: MemoryFact) -> None:
        emb_bytes: Optional[bytes] = None
        if fact.embedding is not None:
            emb_bytes = zlib.compress(fact.embedding.astype(np.float32).tobytes())
        fid = hashlib.sha256(fact.text.encode()).hexdigest()
        with self._pool.transaction():
            self._pool.get().execute(
                "INSERT OR REPLACE INTO facts VALUES (?,?,?,?,?,?)",
                (fid, fact.text, emb_bytes,
                 fact.created_at, fact.session_id, fact.importance),
            )

    def get_all(self) -> List[MemoryFact]:
        conn = self._pool.get()
        rows = conn.execute(
            "SELECT * FROM facts ORDER BY created_at DESC"
        ).fetchall()
        facts = []
        for row in rows:
            emb: Optional[np.ndarray] = None
            if row["embedding"]:
                raw = zlib.decompress(row["embedding"])
                emb = np.frombuffer(raw, dtype=np.float32).copy()
            facts.append(MemoryFact(
                text=row["text"],
                embedding=emb,
                created_at=row["created_at"],
                session_id=row["session_id"],
                importance=row["importance"],
            ))
        return facts

    def delete_expired(self, ttl: float) -> int:
        cutoff = time.time() - ttl
        with self._pool.transaction():
            cur = self._pool.get().execute(
                "DELETE FROM facts WHERE created_at < ?", (cutoff,)
            )
        return cur.rowcount


class RedisMemoryBackend(_MemoryBackend):
    def __init__(self, redis_url: str, ttl: int) -> None:
        if not _REDIS_AVAILABLE:
            raise ConfigurationError("redis-py not installed")
        self._r   = _redis_mod.from_url(redis_url, decode_responses=False)
        self._ttl = ttl
        self._key = "navigator:memory:facts"

    def add(self, fact: MemoryFact) -> None:
        emb_b64 = ""
        if fact.embedding is not None:
            compressed = zlib.compress(fact.embedding.astype(np.float32).tobytes())
            emb_b64    = base64.b64encode(compressed).decode("ascii")
        entry = {
            "text":       fact.text,
            "ts":         fact.created_at,
            "session":    fact.session_id,
            "importance": fact.importance,
            "emb":        emb_b64,
        }
        self._r.zadd(self._key, {json.dumps(entry): fact.created_at})
        self._r.expire(self._key, self._ttl)

    def get_all(self) -> List[MemoryFact]:
        raw = self._r.zrevrange(self._key, 0, -1, withscores=True)
        facts: List[MemoryFact] = []
        for data, score in raw:
            try:
                entry = json.loads(data)
                emb: Optional[np.ndarray] = None
                if entry.get("emb"):
                    compressed = base64.b64decode(entry["emb"])
                    raw_bytes  = zlib.decompress(compressed)
                    emb        = np.frombuffer(raw_bytes, dtype=np.float32).copy()
                facts.append(MemoryFact(
                    text=entry["text"],
                    embedding=emb,
                    created_at=score,
                    session_id=entry.get("session", ""),
                    importance=entry.get("importance", 1.0),
                ))
            except Exception:
                continue
        return facts

    def delete_expired(self, ttl: float) -> int:
        cutoff = time.time() - ttl
        return self._r.zremrangebyscore(self._key, "-inf", cutoff)


class LongTermMemory:
    """Persistent memory with real-embedding retrieval (cosine similarity)."""

    def __init__(self, cfg: Config, session_id: str, embed_provider: EmbeddingProvider) -> None:
        self._cfg        = cfg
        self._session_id = session_id
        self._embed      = embed_provider
        self._lock       = threading.Lock()

        if cfg.memory_backend == "redis":
            self._backend: _MemoryBackend = RedisMemoryBackend(
                cfg.redis_url, cfg.memory_ttl_seconds
            )
        else:
            self._backend = SQLiteMemoryBackend(cfg.session_db_path)

        self._extract_sem = threading.Semaphore(cfg.fact_extract_concurrency)

    def add(self, text: str, importance: float = 1.0) -> None:
        if not text:
            return
        emb  = self._embed.embed(text)
        fact = MemoryFact(
            text=text,
            embedding=emb,
            created_at=time.time(),
            session_id=self._session_id,
            importance=importance,
        )
        with self._lock:
            self._backend.add(fact)
        _log("memory", f"Stored [{importance:.1f}]: {text[:80]}")

    def retrieve(self, query: str, top_k: Optional[int] = None) -> List[str]:
        k = top_k or self._cfg.memory_top_k
        with self._lock:
            facts = self._backend.get_all()
        if not facts:
            return []

        q_emb   = self._embed.embed(query)
        scored: List[Tuple[float, MemoryFact]] = []

        for fact in facts:
            if fact.embedding is not None and len(fact.embedding) > 0:
                sim = EmbeddingProvider.cosine(q_emb, fact.embedding) * fact.importance
            else:
                age_hours = (time.time() - fact.created_at) / 3600.0
                sim       = max(0.0, 1.0 - age_hours / 24.0) * fact.importance
            scored.append((sim, fact))

        scored.sort(key=lambda x: x[0], reverse=True)
        return [f.text for _, f in scored[:k]]

    def summary(self) -> str:
        with self._lock:
            facts = self._backend.get_all()
        if not facts:
            return "No long-term facts stored."
        return "\n".join(f"• {f.text}" for f in facts[:8])

    def prune_expired(self) -> None:
        removed = self._backend.delete_expired(self._cfg.memory_ttl_seconds)
        if removed:
            _log("memory", f"Pruned {removed} expired facts")

    @property
    def extract_semaphore(self) -> threading.Semaphore:
        return self._extract_sem

# SESSION STATE STORE  (v5 canonical: SQLite WAL, full history table)
# ─────────────────────────────────────────────────────────────────────────────
class SessionStore:
    def __init__(self, db_path: str) -> None:
        self._pool = _SQLitePool(db_path)
        self._init_db()

    def _init_db(self) -> None:
        conn = self._pool.get()
        conn.executescript("""
            CREATE TABLE IF NOT EXISTS sessions (
                session_id  TEXT PRIMARY KEY,
                task        TEXT,
                status      TEXT DEFAULT 'running',
                created_at  REAL,
                updated_at  REAL,
                node_id     TEXT,
                step_count  INTEGER DEFAULT 0,
                result      TEXT
            );
            CREATE TABLE IF NOT EXISTS history (
                id          INTEGER PRIMARY KEY AUTOINCREMENT,
                session_id  TEXT NOT NULL,
                role        TEXT NOT NULL,
                content     TEXT NOT NULL,
                step        INTEGER,
                ts          REAL,
                FOREIGN KEY (session_id) REFERENCES sessions(session_id)
            );
            CREATE INDEX IF NOT EXISTS idx_hist_session ON history(session_id);
        """)

    def create_session(self, task: str, node_id: str) -> str:
        sid = uuid.uuid4().hex
        now = time.time()
        self._pool.get().execute(
            "INSERT INTO sessions VALUES (?,?,?,?,?,?,?,?)",
            (sid, task, "running", now, now, node_id, 0, None),
        )
        return sid

    def update_session(
        self,
        session_id: str,
        status: Optional[str] = None,
        step_count: Optional[int] = None,
        result: Optional[str] = None,
    ) -> None:
        now    = time.time()
        fields = ["updated_at = ?"]
        vals   = [now]
        if status is not None:
            fields.append("status = ?")
            vals.append(status)
        if step_count is not None:
            fields.append("step_count = ?")
            vals.append(step_count)
        if result is not None:
            fields.append("result = ?")
            vals.append(result)
        vals.append(session_id)
        self._pool.get().execute(
            f"UPDATE sessions SET {', '.join(fields)} WHERE session_id = ?",
            vals,
        )

    def append_history(self, session_id: str, role: str, content: str, step: int) -> None:
        self._pool.get().execute(
            "INSERT INTO history (session_id, role, content, step, ts) VALUES (?,?,?,?,?)",
            (session_id, role, content, step, time.time()),
        )

    def load_history(self, session_id: str) -> List[Dict[str, Any]]:
        rows = self._pool.get().execute(
            "SELECT role, content FROM history WHERE session_id = ? ORDER BY id",
            (session_id,),
        ).fetchall()
        return [{"role": row["role"], "content": row["content"]} for row in rows]


# ─────────────────────────────────────────────────────────────────────────────
# FRAME DEDUPLICATION HELPER
# ─────────────────────────────────────────────────────────────────────────────
def _phash_similarity(a: np.ndarray, b: np.ndarray) -> float:
    """Normalised perceptual-hash similarity in [0, 1]."""
    if a is None or b is None:
        return 0.0
    diff = np.count_nonzero(a.astype(np.int16) - b.astype(np.int16))
    total = a.size
    return 1.0 - diff / max(total, 1)


def _compute_phash(frame: np.ndarray, hash_size: int = 16) -> Optional[np.ndarray]:
    """Classic perceptual hash (DCT-based) for frame deduplication."""
    try:
        gray    = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        resized = cv2.resize(gray, (hash_size * 4, hash_size * 4))
        dct     = cv2.dct(resized.astype(np.float32))
        dct_low = dct[:hash_size, :hash_size]
        med     = np.median(dct_low)
        return (dct_low > med).astype(np.uint8)
    except Exception:
        return None

# OCR EXTRACTOR  (v5 canonical: tesseract / easyocr / auto)
# ─────────────────────────────────────────────────────────────────────────────
class OCRExtractor:
    def __init__(self, cfg: Config) -> None:
        self._cfg = cfg

    def extract(self, frame: np.ndarray) -> str:
        if not cfg_ocr_enabled(self._cfg):
            return ""
        backend = self._cfg.ocr_backend.lower()
        text = ""
        try:
            if backend in ("tesseract", "auto") and _TESSERACT_AVAILABLE:
                text = self._tesseract(frame)
            elif backend in ("easyocr", "auto") and _EASYOCR_AVAILABLE:
                text = self._easyocr(frame)
        except Exception as exc:
            _log("ocr", f"OCR extraction failed: {exc}", "warning")
        return text[:self._cfg.ocr_max_chars]

    @staticmethod
    def _tesseract(frame: np.ndarray) -> str:
        rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        return pytesseract.image_to_string(rgb)

    @staticmethod
    def _easyocr(frame: np.ndarray) -> str:
        global _easyocr_reader
        if _easyocr_reader is None:
            _easyocr_reader = easyocr.Reader(["en"], gpu=False)
        rgb     = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        results = _easyocr_reader.readtext(rgb)
        return " ".join(r[1] for r in results)

# SCREEN BUFFER  (v5 canonical: async capture, frame dedup, dynamic sampling)
# ─────────────────────────────────────────────────────────────────────────────
class ScreenBuffer:
    """
    Continuously captures frames from the configured monitor at capture_fps.
    Deduplicates consecutive static frames via perceptual hash.
    Provides get_sampled_frames() for the agent loop with dynamic frame count.
    """

    def __init__(self, cfg: Config) -> None:
        self.cfg      = cfg
        self._frames: deque = deque(maxlen=cfg.capture_fps * cfg.buffer_seconds)
        self._lock    = threading.Lock()
        self._thread: Optional[threading.Thread] = None
        self._stop    = threading.Event()
        self._mss     = mss.mss()

        mon_idx = min(cfg.monitor, len(self._mss.monitors) - 1)
        mon     = self._mss.monitors[mon_idx]
        self.screen_w = mon["width"]
        self.screen_h = mon["height"]

        self._last_hash: Optional[np.ndarray] = None

    @property
    def screen_size(self) -> Tuple[int, int]:
        return self.screen_w, self.screen_h

    def start(self) -> None:
        self._stop.clear()
        self._thread = threading.Thread(
            target=self._capture_loop, daemon=True, name="Nav-ScreenCapture"
        )
        self._thread.start()
        _log("screen", "Screen capture started")

    def stop(self) -> None:
        self._stop.set()
        if self._thread:
            self._thread.join(timeout=2.0)
        _log("screen", "Screen capture stopped")

    def _capture_loop(self) -> None:
        interval = 1.0 / max(self.cfg.capture_fps, 1)
        mon_idx  = min(self.cfg.monitor, len(self._mss.monitors) - 1)
        monitor  = self._mss.monitors[mon_idx]
        while not self._stop.is_set():
            t0 = time.monotonic()
            try:
                shot  = self._mss.grab(monitor)
                frame = np.array(shot)
                if frame.shape[2] == 4:
                    frame = cv2.cvtColor(frame, cv2.COLOR_BGRA2BGR)

                if self.cfg.enable_frame_dedup:
                    ph = _compute_phash(frame)
                    if ph is not None and self._last_hash is not None:
                        if _phash_similarity(ph, self._last_hash) >= self.cfg.dedup_threshold:
                            time.sleep(max(0.0, interval - (time.monotonic() - t0)))
                            continue
                    self._last_hash = ph

                with self._lock:
                    self._frames.append(frame)
            except Exception as exc:
                _log("screen", f"Capture error: {exc}", "warning")
            time.sleep(max(0.0, interval - (time.monotonic() - t0)))

    def get_latest_frame(self) -> Optional[np.ndarray]:
        with self._lock:
            return self._frames[-1].copy() if self._frames else None

    def get_sampled_frames(self) -> Tuple[List[str], bool]:
        """
        Returns (list_of_b64_jpeg_strings, is_static).
        Dynamic frame count: fewer frames on static screens, more on moving ones.
        """
        with self._lock:
            frames = list(self._frames)

        if not frames:
            return [], True

        # Detect motion by comparing first/last frame in buffer
        is_static = True
        if len(frames) >= 2:
            ph_first = _compute_phash(frames[0])
            ph_last  = _compute_phash(frames[-1])
            if ph_first is not None and ph_last is not None:
                sim       = _phash_similarity(ph_first, ph_last)
                is_static = sim >= self.cfg.static_threshold

        n = self.cfg.min_sample_frames if is_static else self.cfg.max_sample_frames
        step     = max(1, len(frames) // n)
        selected = frames[::step][:n]

        b64_frames = []
        for f in selected:
            ok, buf = cv2.imencode(
                ".jpg", f, [cv2.IMWRITE_JPEG_QUALITY, self.cfg.encode_quality]
            )
            if ok:
                b64_frames.append(base64.b64encode(buf.tobytes()).decode("ascii"))

        return b64_frames, is_static

# SAFETY GUARD  (v5 canonical: semantic check + user confirm)
# ─────────────────────────────────────────────────────────────────────────────
class SafetyGuard:
    """
    Two-layer safety check:
      1. Semantic classifier via safety_model API.
      2. Optional human confirmation prompt for high-risk actions.
    """

    _HIGH_RISK = {ActionType.KEY, ActionType.HOTKEY, ActionType.TYPE}
    _HIGH_RISK_PATTERNS = re.compile(
        r"\b(delete|remove|format|shutdown|reboot|kill|drop|truncate|rm\s+-rf)\b",
        re.IGNORECASE,
    )

    _SAFETY_SYSTEM = (
        "You are a safety classifier for a computer-use AI agent. "
        "Assess whether the proposed action is SAFE or UNSAFE given the current context. "
        "An action is UNSAFE if it could cause irreversible data loss, system damage, "
        "security breaches, or violates the user's stated task intent. "
        'Respond ONLY with valid JSON: {"safe": true|false, "reason": "<brief reason>"}'
    )

    def __init__(self, api_key: str, cfg: Config) -> None:
        self._cfg    = cfg
        self._client = openai.OpenAI(
            api_key=api_key,
            base_url="https://api.x.ai/v1",
            timeout=15.0,
            max_retries=0,
        )

    def check(self, action: Action, context: str, auto_confirm: bool) -> None:
        """Raises SafetyError if the action should be blocked."""
        action_str = json.dumps(asdict(action), default=str)

        # Fast local pattern check
        if self._HIGH_RISK_PATTERNS.search(action_str):
            reason = "Action matches high-risk keyword pattern"
            if not auto_confirm:
                if not self._prompt_user_confirm(action, reason):
                    raise SafetyError(f"User rejected high-risk action: {reason}")

        # Semantic API check
        try:
            raw = self._client.chat.completions.create(
                model=self._cfg.safety_model,
                messages=[
                    {"role": "system", "content": self._SAFETY_SYSTEM},
                    {
                        "role": "user",
                        "content": (
                            f"Context: {context}\n"
                            f"Proposed action: {action_str}"
                        ),
                    },
                ],
                max_tokens=100,
                temperature=0.0,
            ).choices[0].message.content

            cleaned = re.sub(r"```(?:json)?|```", "", raw).strip()
            data    = json.loads(cleaned)
            if not data.get("safe", True):
                reason = data.get("reason", "Safety classifier flagged action")
                if auto_confirm:
                    _log("safety", f"Auto-confirmed unsafe action: {reason}", "warning")
                else:
                    if not self._prompt_user_confirm(action, reason):
                        raise SafetyError(f"Action blocked by safety classifier: {reason}")
        except (json.JSONDecodeError, KeyError):
            pass   # classifier parse error → allow action
        except SafetyError:
            raise
        except Exception as exc:
            _log("safety", f"Safety check failed (non-critical): {exc}", "warning")

    def _prompt_user_confirm(self, action: Action, reason: str) -> bool:
        print(f"\n⚠️  HIGH-RISK ACTION DETECTED")
        print(f"   Reason : {reason}")
        print(f"   Action : {action.type.value} — {action.text or action.key or ''}")
        answer = input("   Allow? [y/N] ").strip().lower()
        return answer in ("y", "yes")


# ─────────────────────────────────────────────────────────────────────────────
# API RETRY WRAPPER  (v5 canonical: jitter backoff, circuit breaker aware)
# ─────────────────────────────────────────────────────────────────────────────
@contextmanager
def _noop_ctx():
    yield


def api_call_with_retry(
    fn:              Callable,
    *args:           Any,
    max_retries:     int   = 5,
    base_backoff:    float = 1.0,
    max_backoff:     float = 60.0,
    rate_limiter:    Optional[AdaptiveRateLimiter] = None,
    circuit_breaker: Optional[CircuitBreaker]      = None,
    **kwargs:        Any,
) -> Any:
    last_exc: Optional[Exception] = None
    for attempt in range(max_retries + 1):
        if rate_limiter:
            rate_limiter.acquire()
        try:
            ctx = circuit_breaker.guard() if circuit_breaker else _noop_ctx()
            with ctx:
                result = fn(*args, **kwargs)
            if rate_limiter:
                rate_limiter.on_success()
            return result
        except RateLimitError as exc:
            if rate_limiter:
                rate_limiter.on_rate_limited()
            wait = min(
                base_backoff * (2 ** attempt) + random.uniform(0, 1),
                max_backoff,
            )
            _log("api",
                 f"Rate limited (attempt {attempt+1}/{max_retries}), "
                 f"retry in {wait:.1f}s", "warning")
            last_exc = exc
        except (APIConnectionError, APIStatusError) as exc:
            if isinstance(exc, APIStatusError) and exc.status_code < 500:
                raise APIError(f"Non-retryable API error: {exc}") from exc
            wait = min(
                base_backoff * (2 ** attempt) + random.uniform(0, 1),
                max_backoff,
            )
            _log("api",
                 f"API error (attempt {attempt+1}/{max_retries}), "
                 f"retry in {wait:.1f}s: {exc}", "warning")
            last_exc = exc
        except CircuitOpenError:
            raise
        except Exception as exc:
            raise APIError(f"Unexpected API failure: {exc}") from exc

        if attempt < max_retries:
            time.sleep(wait)   # type: ignore[possibly-undefined]

    raise APIError(f"API call failed after {max_retries} retries") from last_exc

# HISTORY COMPRESSOR  (v6 improvement: uses compression_model not vision model)
# ─────────────────────────────────────────────────────────────────────────────
class HistoryCompressor:
    """
    Compresses the raw conversation history into a short textual summary every
    N steps, replacing verbose turn-by-turn records with a compact description.
    Uses cfg.compression_model (default: grok-3-mini) — not the expensive
    vision model.
    """

    _COMPRESS_SYSTEM = (
        "You are a concise summariser for an AI agent's action log. "
        "Given a sequence of agent steps, produce a 3-6 sentence plain-English "
        "summary capturing: what was attempted, what succeeded, what failed, "
        "and the current state. Be specific about UI elements and actions. "
        "Respond ONLY with the summary text — no JSON, no markdown."
    )

    def __init__(self, cfg: Config, client: openai.OpenAI, rate: AdaptiveRateLimiter) -> None:
        self._cfg        = cfg
        self._client     = client
        self._rate       = rate
        self._model_name = cfg.compression_model   # v6: cheaper model

    def should_compress(self, step: int, history: List[Dict]) -> bool:
        return (
            self._cfg.history_compress_enabled
            and step > 0
            and step % self._cfg.history_compress_every == 0
            and len(history) > self._cfg.history_compress_keep + 2
        )

    def compress(self, history: List[Dict], task: str) -> List[Dict]:
        keep        = self._cfg.history_compress_keep
        to_compress = history[:-keep] if keep > 0 else history
        tail        = history[-keep:] if keep > 0 else []

        flat_text = ""
        for msg in to_compress:
            role    = msg.get("role", "?")
            content = msg.get("content", "")
            if isinstance(content, list):
                content = " ".join(
                    c.get("text", "") for c in content if isinstance(c, dict)
                )
            flat_text += f"[{role.upper()}]: {str(content)[:400]}\n"

        try:
            summary: str = api_call_with_retry(
                lambda: self._client.chat.completions.create(
                    model=self._model_name,
                    messages=[
                        {"role": "system", "content": self._COMPRESS_SYSTEM},
                        {"role": "user",   "content": f"Task: {task}\n\n{flat_text}"},
                    ],
                    max_tokens=250,
                    temperature=0.0,
                ).choices[0].message.content,
                max_retries=1,
                rate_limiter=self._rate,
            )
        except Exception as exc:
            _log("compress", f"History compression failed: {exc}", "warning")
            return history

        compressed_msg = {
            "role":    "assistant",
            "content": f"[COMPRESSED SUMMARY — steps so far]\n{summary.strip()}",
        }
        _log("compress", f"History compressed: {len(to_compress)} turns → 1 summary block")
        return [compressed_msg] + tail

class ReflectionEngine:
    """
    Every reflection_interval steps the agent evaluates its own progress,
    detects repetitive loops, and receives a corrective strategy hint.
    Uses cfg.reflection_model (default: grok-3-mini).
    """

    _REFLECT_SYSTEM = (
        "You are an autonomous self-critique module for an AI agent. "
        "Review the agent's recent action history and evaluate:\n"
        "1. Is the agent making progress toward the task?\n"
        "2. Is it stuck in a repetitive loop? (list repeated actions)\n"
        "3. What is the most productive next action or strategy?\n\n"
        "Respond ONLY with valid JSON:\n"
        "{\n"
        '  "progress": "good|slow|stuck|regressing",\n'
        '  "loop_detected": true|false,\n'
        '  "repeated_action": "<description or null>",\n'
        '  "recommendation": "<1-2 sentence corrective instruction>"\n'
        "}"
    )

    def __init__(self, cfg: Config, client: openai.OpenAI, rate: AdaptiveRateLimiter) -> None:
        self._cfg        = cfg
        self._client     = client
        self._rate       = rate
        self._model_name = cfg.reflection_model   # v6: cheaper model

    def should_reflect(self, step: int) -> bool:
        return (
            self._cfg.reflection_enabled
            and step > 0
            and step % self._cfg.reflection_interval == 0
        )

    def reflect(self, task: str, history: List[Dict], step: int) -> Optional[str]:
        recent = history[-min(len(history), self._cfg.reflection_interval * 2):]
        flat   = ""
        for msg in recent:
            role    = msg.get("role", "?")
            content = msg.get("content", "")
            if isinstance(content, list):
                content = " ".join(
                    c.get("text", "") for c in content if isinstance(c, dict)
                )
            flat += f"[{role.upper()}]: {str(content)[:300]}\n"

        try:
            raw: str = api_call_with_retry(
                lambda: self._client.chat.completions.create(
                    model=self._model_name,
                    messages=[
                        {"role": "system", "content": self._REFLECT_SYSTEM},
                        {
                            "role": "user",
                            "content": (
                                f"Task: {task}\nCurrent step: {step}\n\n"
                                f"Recent history:\n{flat}"
                            ),
                        },
                    ],
                    max_tokens=200,
                    temperature=0.0,
                ).choices[0].message.content,
                max_retries=1,
                rate_limiter=self._rate,
            )
            cleaned    = re.sub(r"```(?:json)?|```", "", raw).strip()
            data       = json.loads(cleaned)
            progress   = data.get("progress", "unknown")
            loop_det   = data.get("loop_detected", False)
            rec        = data.get("recommendation", "")
            rep_action = data.get("repeated_action", "")

            _log("reflect",
                 f"Step {step} reflection → progress={progress}, "
                 f"loop={loop_det}, rec: {rec[:80]}")

            hint = f"[SELF-CRITIQUE at step {step}] Progress: {progress}."
            if loop_det and rep_action:
                hint += f" LOOP DETECTED: {rep_action}. STOP repeating this."
            if rec:
                hint += f" Recommended strategy: {rec}"
            return hint

        except Exception as exc:
            _log("reflect", f"Reflection failed (non-critical): {exc}", "warning")
            return None

# v6: GROUNDING ENGINE (Florence-2, v6.1 hardened parser)
# ─────────────────────────────────────────────────────────────────────────────
class GroundingEngine:
    """
    On-device visual grounding with Florence-2.
    Returns list of {"label": str, "bbox": [x1,y1,x2,y2], "center": (cx,cy)}.

    v6.1 improvements:
      - Robust regex handles punctuation, hyphens, non-ASCII labels.
      - Centre deduplication within ±5 px prevents near-duplicate detections.
      - Hard cap of 20 elements prevents context bloat.
      - Full error resilience: bad coordinate values skipped gracefully.
    """

    def __init__(self, cfg: Config) -> None:
        self.cfg     = cfg
        self.enabled = cfg.grounding_enabled and _FLORENCE_AVAILABLE
        self.model   = None
        self.processor = None
        if self.enabled:
            try:
                self.processor = AutoProcessor.from_pretrained(
                    cfg.grounding_model, trust_remote_code=True
                )
                self.model = AutoModelForCausalLM.from_pretrained(
                    cfg.grounding_model, trust_remote_code=True
                )
                device     = "cuda" if torch.cuda.is_available() else "cpu"
                self.model = self.model.to(device)
                _log("grounding", f"Florence-2 loaded on {device} — visual grounding active")
            except Exception as exc:
                _log("grounding",
                     f"Florence-2 load failed: {exc}. Disabling grounding.", "warning")
                self.enabled = False

    def detect_elements(self, frame: Optional[np.ndarray]) -> List[Dict[str, Any]]:
        if not self.enabled or self.model is None or frame is None:
            return []
        try:
            rgb    = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
            image  = Image.fromarray(rgb)
            prompt = "<OD>"
            inputs = self.processor(
                text=prompt, images=image, return_tensors="pt"
            ).to(self.model.device)
            generated_ids = self.model.generate(
                input_ids=inputs["input_ids"],
                pixel_values=inputs["pixel_values"],
                max_new_tokens=1024,
                num_beams=3,
                do_sample=False,
            )
            generated_text = self.processor.batch_decode(
                generated_ids, skip_special_tokens=False
            )[0]
            return self._parse_od_output(generated_text, image.size)
        except Exception as exc:
            _log("grounding", f"Detection failed: {exc}", "warning")
            return []

    def _parse_od_output(
        self, text: str, image_size: Tuple[int, int]
    ) -> List[Dict[str, Any]]:
        """
        v6.1 hardened parser: robust regex, ±5 px centre dedup, error resilience.
        Florence-2 normalises coordinates to 0-999.
        """
        elements: List[Dict[str, Any]] = []
        # Matches any non-'<' label (handles punctuation, hyphens, etc.)
        pattern = re.compile(r'([^\s<][^<]*?)<loc>(\d+)\s+(\d+)\s+(\d+)\s+(\d+)</loc>')
        w, h    = image_size
        seen: set = set()

        for match in pattern.finditer(text):
            label = match.group(1).strip()
            try:
                y1, x1, y2, x2 = map(int, match.group(2, 3, 4, 5))
                x1 = int(x1 / 999 * w)
                y1 = int(y1 / 999 * h)
                x2 = int(x2 / 999 * w)
                y2 = int(y2 / 999 * h)
                cx, cy = (x1 + x2) // 2, (y1 + y2) // 2
                key = (cx // 5, cy // 5)   # ±5 px deduplication bucket
                if key in seen:
                    continue
                seen.add(key)
                elements.append({
                    "label": label,
                    "bbox":  [x1, y1, x2, y2],
                    "center": (cx, cy),
                })
            except ValueError:
                continue   # skip malformed coordinate tuples

        return elements[:20]   # cap to prevent context bloat

# STUCK DETECTOR  (v5 canonical: consecutive-frame similarity threshold)
# ─────────────────────────────────────────────────────────────────────────────
class StuckDetector:
    """
    Detects when the agent is stuck by comparing consecutive screen frames.
    If phash similarity exceeds stuck_threshold for stuck_consecutive frames,
    raises the stuck flag.
    """

    def __init__(self, threshold: float = 0.96, consecutive: int = 3) -> None:
        self._threshold   = threshold
        self._consecutive = consecutive
        self._count       = 0
        self._last_hash:  Optional[np.ndarray] = None

    def check(self, frame: np.ndarray) -> bool:
        ph = _compute_phash(frame)
        if ph is None:
            return False
        if self._last_hash is not None:
            sim = _phash_similarity(ph, self._last_hash)
            if sim >= self._threshold:
                self._count += 1
            else:
                self._count = 0
        self._last_hash = ph
        return self._count >= self._consecutive

    def reset(self) -> None:
        self._count = 0
        self._last_hash = None

# DISTRIBUTED TASK QUEUE  (v5 canonical: Redis-backed, heartbeat, worker mode)
# ─────────────────────────────────────────────────────────────────────────────
class DistributedTaskQueue:
    """
    Redis-backed task queue for multi-node agent deployment.
    Gracefully degrades to no-op when Redis is unavailable.
    """

    def __init__(self, cfg: Config) -> None:
        self._cfg = cfg
        self._r: Optional[Any] = None
        if _REDIS_AVAILABLE:
            try:
                self._r = _redis_mod.from_url(cfg.redis_url, socket_timeout=3.0)
                self._r.ping()
                _log("queue", f"Connected to Redis at {cfg.redis_url}")
            except Exception as exc:
                _log("queue", f"Redis unavailable: {exc} — queue disabled", "warning")
                self._r = None

    def push_task(self, task_data: Dict[str, Any]) -> bool:
        if self._r is None:
            return False
        try:
            self._r.rpush(self._cfg.task_queue_key, json.dumps(task_data))
            return True
        except Exception as exc:
            _log("queue", f"push_task failed: {exc}", "error")
            return False

    def pop_task(self, timeout: float = 5.0) -> Optional[Dict[str, Any]]:
        if self._r is None:
            return None
        try:
            result = self._r.blpop(self._cfg.task_queue_key, timeout=int(timeout))
            if result:
                _, raw = result
                return json.loads(raw)
        except Exception as exc:
            _log("queue", f"pop_task failed: {exc}", "error")
        return None

    def push_result(self, result_data: Dict[str, Any]) -> bool:
        if self._r is None:
            return False
        try:
            self._r.rpush(self._cfg.result_queue_key, json.dumps(result_data))
            self._r.expire(self._cfg.result_queue_key, 86400)
            return True
        except Exception as exc:
            _log("queue", f"push_result failed: {exc}", "error")
            return False

    def heartbeat(self, node_id: str, status: str) -> None:
        if self._r is None:
            return
        try:
            key = f"navigator:node:{node_id}"
            self._r.hset(key, mapping={
                "status":     status,
                "updated_at": datetime.now(timezone.utc).isoformat(),
            })
            self._r.expire(key, int(self._cfg.heartbeat_interval * 3))
        except Exception:
            pass

    def _build_messages(
        self,
        task:               str,
        frames_b64:         List[str],
        history:            List[Dict],
        screen_size:        Tuple[int, int],
        relevant_memory:    List[str],
        ocr_text:           str,
        reflection_hint:    Optional[str],
        grounding_elements: List[Dict],     # v6

        mem_block = ""

    def call(
        self,
        task:         str,
        frames_b64:   List[str],
        history:      List[Dict],
        screen_size:  Tuple[int, int],
        step:         int,
        latest_frame: Optional[np.ndarray] = None,
    ) -> Tuple[Action, str, List[Dict]]:
        """
        Returns (action, raw_response, updated_history).
        History may have been compressed or annotated with reflection.
        """

            history = self.compressor.compress(history, task)

        reflection_hint: Optional[str] = None
        if self.reflection.should_reflect(step):
            reflection_hint = self.reflection.reflect(task, history, step)

        ocr_text = ""
        if latest_frame is not None and cfg_ocr_enabled(self.cfg):
            ocr_text = self.ocr.extract(latest_frame)
            if ocr_text:
                _log("think", f"OCR grounding ({len(ocr_text)} chars)", "debug")

        # v6: run Florence-2 grounding on the latest frame
        grounding_elements: List[Dict] = (
            self.grounding.detect_elements(latest_frame)
            if latest_frame is not None else []
        )
        if grounding_elements:
            _log("think", f"Grounding: {len(grounding_elements)} elements detected", "debug")

        relevant = self.memory.retrieve(task)
        messages = self._build_messages(
            task, frames_b64, history, screen_size,
            relevant, ocr_text, reflection_hint, grounding_elements,
        )

        raw_content: str = api_call_with_retry(
            lambda: self.client.chat.completions.create(
                model=self.cfg.model,
                messages=messages,
                max_tokens=self.cfg.max_tokens,
                temperature=self.cfg.temperature,
            ).choices[0].message.content,
            max_retries=self.cfg.api_max_retries,
            base_backoff=self.cfg.api_base_backoff,
            max_backoff=self.cfg.api_max_backoff,
            rate_limiter=self.rate,
            circuit_breaker=self._breaker,
        )

        try:
            cleaned     = re.sub(r"```(?:json)?|```", "", raw_content).strip()
            action_dict = json.loads(cleaned)
        except json.JSONDecodeError as exc:
            _log("think",
                 f"JSON parse error: {exc} — raw: {raw_content[:200]}", "error")
            return (
                Action(type=ActionType.FAILED, message=f"JSON parse error: {exc}"),
                raw_content,
                history,
            )

        action = Action.parse(action_dict)

        self.audit.write("action_proposed", {
            "step":   step,
            "action": asdict(action),
            "sig":    action.sign(self.cfg.audit_hmac_key),
        })

        # v7 fix: fact extraction uses mini model, not vision model
        threading.Thread(
            target=self._extract_and_store_fact,
            args=(task, raw_content, step),
            daemon=True,
            name=f"Nav-FactExtract-step{step}",
        ).start()

        return action, raw_content, history

    def _extract_and_store_fact(self, task: str, step_content: str, step: int) -> None:
        acquired = self.memory.extract_semaphore.acquire(blocking=False)
        if not acquired:
            return
        try:
            resp: str = api_call_with_retry(
                lambda: self.client.chat.completions.create(
                    model=self.cfg.fact_extract_model,   # v7: mini model
                    messages=[
                        {"role": "system", "content": self._FACT_EXTRACT_PROMPT},
                        {
                            "role": "user",
                            "content": (
                                f"Task: {task}\n"
                                f"Step {step}: {step_content[:500]}"
                            ),
                        },
                    ],
                    max_tokens=60,
                    temperature=0.0,
                ).choices[0].message.content,
                max_retries=1,
                rate_limiter=self.rate,
            )
            cleaned = re.sub(r"```(?:json)?|```", "", resp).strip()
            data    = json.loads(cleaned)
            fact    = data.get("key_fact")
            if fact:
                self.memory.add(fact, importance=0.8)
        except Exception:
            pass
        finally:
            self.memory.extract_semaphore.release()


# ─────────────────────────────────────────────────────────────────────────────
# ACTION EXECUTOR  (v6+v6.1: UndoStack + dHash visual-change detection)
# ─────────────────────────────────────────────────────────────────────────────
class ActionExecutor:
    MOVE_TOLERANCE = 15

    def __init__(self, cfg: Config, screen: ScreenBuffer, dry_run: bool = False) -> None:
        self.cfg     = cfg
        self.screen  = screen
        self.dry_run = dry_run
        self.undo    = UndoStack()           # v6
        self._mss    = mss.mss()             # dedicated capture for dHash
        pyautogui.FAILSAFE = True
        pyautogui.PAUSE    = 0.0

    # ── v6.1: perceptual hash of the current screen ───────────────────────────
    def _get_perceptual_hash(self) -> Optional[int]:
        """Capture current monitor and return dHash integer."""
        try:
            monitors  = self._mss.monitors
            mon_idx   = min(self.cfg.monitor, len(monitors) - 1)
            monitor   = monitors[mon_idx]
            shot      = self._mss.grab(monitor)
            frame     = np.array(shot)
            if frame.shape[2] == 4:
                frame = cv2.cvtColor(frame, cv2.COLOR_BGRA2BGR)
            return dhash(frame)
        except Exception as exc:
            _log("exec", f"Perceptual hash capture failed: {exc}", "warning")
            return None

    def execute(self, action: Action) -> Tuple[bool, bool]:
        """
        Execute action and return (success, visual_changed).

        visual_changed = False signals the NavigatorAgent.run() loop to inject
        a RECOVERY TRIGGER into history so the model knows its action had no
        effect and must choose a different strategy.
        """
        if self.dry_run:
            _log("exec",
                 f"[DRY-RUN] {action.type.value} @ ({action.x},{action.y}) '{action.text}'")
            return True, True

        # Capture pre-action hash for mutating action types
        mutating = action.type in (
            ActionType.CLICK, ActionType.DOUBLE_CLICK, ActionType.RIGHT_CLICK,
            ActionType.TYPE, ActionType.HOTKEY, ActionType.SCROLL,
        )
        pre_hash = self._get_perceptual_hash() if mutating else None

        try:
            # v6: snapshot state before risky actions for UNDO support
            if action.type in (
                ActionType.CLICK, ActionType.DOUBLE_CLICK, ActionType.RIGHT_CLICK,
                ActionType.MOVE_TO, ActionType.TYPE, ActionType.HOTKEY,
            ):
                current_mouse = pyautogui.position()
                current_clip  = pyperclip.paste() if _PYPERCLIP_AVAILABLE else None
                self.undo.push(current_mouse, current_clip)

            match action.type:
                case ActionType.CLICK:
                    pyautogui.moveTo(action.x, action.y,
                                     duration=self.cfg.mouse_move_duration)
                    pyautogui.click()

                case ActionType.DOUBLE_CLICK:
                    pyautogui.moveTo(action.x, action.y,
                                     duration=self.cfg.mouse_move_duration)
                    pyautogui.doubleClick()

                case ActionType.RIGHT_CLICK:
                    pyautogui.moveTo(action.x, action.y,
                                     duration=self.cfg.mouse_move_duration)
                    pyautogui.rightClick()

                case ActionType.MOVE_TO:
                    tx, ty = self._resolve_move_target(action)
                    pyautogui.moveTo(tx, ty, duration=self.cfg.mouse_move_duration)
                    _log("exec",
                         f"MOVE_TO ({tx},{ty})"
                         + (f" [{action.element}]" if action.element else ""))

                case ActionType.TYPE:
                    pyautogui.write(action.text or "", interval=self.cfg.type_interval)

                case ActionType.KEY:
                    pyautogui.press(action.key)

                case ActionType.HOTKEY:
                    pyautogui.hotkey(*(action.keys or []))

                case ActionType.SCROLL:
                    direction_map = {"up": 1, "down": -1, "left": -1, "right": 1}
                    clicks = (action.amount or 3) * direction_map.get(
                        action.direction or "down", -1
                    )
                    pyautogui.scroll(clicks, x=action.x, y=action.y)

                case ActionType.WAIT:
                    time.sleep(action.duration or 1.0)

                case ActionType.CLIPBOARD:
                    if _PYPERCLIP_AVAILABLE:
                        pyperclip.copy(action.text or "")

                case ActionType.UNDO:   # v6
                    state = self.undo.pop()
                    if state:
                        pyautogui.moveTo(*state["mouse"])
                        if state["clipboard"] is not None and _PYPERCLIP_AVAILABLE:
                            pyperclip.copy(state["clipboard"])
                        _log("exec", "UNDO executed — restored mouse position and clipboard")
                    return True, True   # undo always "changed" the state

                case ActionType.SCREENSHOT | ActionType.DONE | ActionType.FAILED:
                    pass

                case _:
                    _log("exec", f"Unknown action type: {action.type}", "warning")
                    return False, True

            if self.cfg.verify_clicks and action.type in (
                ActionType.CLICK, ActionType.DOUBLE_CLICK, ActionType.RIGHT_CLICK
            ):
                self._verify_click(action)

            # v6.1: compare post-action hash; flag no-change for recovery
            visual_changed = True
            if mutating and pre_hash is not None:
                post_hash      = self._get_perceptual_hash()
                visual_changed = (post_hash is None or pre_hash != post_hash)
                if not visual_changed:
                    _log("exec",
                         f"NO VISUAL CHANGE after {action.type.value} "
                         "— recovery trigger will fire", "warning")

            return True, visual_changed

        except pyautogui.FailSafeException:
            raise SafetyError("PyAutoGUI failsafe triggered — mouse in corner")
        except Exception as exc:
            _log("exec", f"Action execution failed: {exc}", "error")
            return False, True

    def _resolve_move_target(self, action: Action) -> Tuple[int, int]:
        tx = action.x or (self.screen.screen_w // 2)
        ty = action.y or (self.screen.screen_h // 2)
        tx += random.randint(-self.MOVE_TOLERANCE // 3, self.MOVE_TOLERANCE // 3)
        ty += random.randint(-self.MOVE_TOLERANCE // 3, self.MOVE_TOLERANCE // 3)
        tx = max(0, min(tx, self.screen.screen_w - 1))
        ty = max(0, min(ty, self.screen.screen_h - 1))
        return tx, ty

    def _verify_click(self, action: Action) -> None:
        time.sleep(0.05)
        cx, cy = pyautogui.position()
        dist   = ((cx - (action.x or 0)) ** 2 + (cy - (action.y or 0)) ** 2) ** 0.5
        if dist > 5:
            _log("exec",
                 f"Click verification mismatch: target=({action.x},{action.y}) "
                 f"actual=({cx},{cy}) dist={dist:.1f}px", "warning")


# ─────────────────────────────────────────────────────────────────────────────
# NAVIGATOR AGENT  (v5 full run loop + v6.1 recovery trigger integrated)
# ─────────────────────────────────────────────────────────────────────────────
class NavigatorAgent:
    def __init__(
        self,
        cfg:          Config,
        dry_run:      bool = False,
        auto_confirm: bool = False,
    ) -> None:
        self.cfg          = cfg
        self.dry_run      = dry_run
        self.auto_confirm = auto_confirm

        session_id      = uuid.uuid4().hex
        self.session_id = session_id

        self.audit = AuditLog(
            cfg.audit_log_path, cfg.audit_hmac_key, cfg.enable_audit_log
        )

        # Shared embedding provider + LRU cache
        self.embed_provider = EmbeddingProvider(cfg)
        self.embed_cache    = EmbeddingCache(maxsize=cfg.embed_cache_size)

        self.memory     = LongTermMemory(cfg, session_id, self.embed_provider)
        self.session_db = SessionStore(cfg.session_db_path)
        self.screen     = ScreenBuffer(cfg)
        self.rate       = AdaptiveRateLimiter(cfg.rate_limit_calls_per_min)
        self.ocr        = OCRExtractor(cfg)

        self.brain    = ThinkAndAct(
            cfg, self.rate, self.audit, self.memory, self.ocr, self.embed_cache
        )
        self.executor   = ActionExecutor(cfg, self.screen, dry_run)
        self.safety     = SafetyGuard(os.environ["XAI_API_KEY"], cfg)
        self.task_queue = DistributedTaskQueue(cfg)
        self.stuck      = StuckDetector(
            cfg.stuck_threshold, cfg.stuck_consecutive
        )

        self._shutdown = threading.Event()
        self._setup_signals()

    def _setup_signals(self) -> None:
        for sig in (signal.SIGINT, signal.SIGTERM):
            signal.signal(sig, self._handle_signal)

    def _handle_signal(self, signum: int, frame: Any) -> None:
        _log("agent", f"Received signal {signum} — initiating graceful shutdown")
        self._shutdown.set()

    def _heartbeat_loop(self, db_session_id: str) -> None:
        while not self._shutdown.is_set():
            try:
                self.task_queue.heartbeat(self.cfg.node_id, "running")
                self.session_db.update_session(db_session_id, status="running")
            except Exception as exc:
                _log("agent", f"Heartbeat error: {exc}", "warning")
            time.sleep(self.cfg.heartbeat_interval)

    def run(self, task: str) -> TaskResult:
        """
        Execute a task end-to-end.  Returns a structured TaskResult.

        All improvements active in this loop:
          • Dynamic frame sampling
          • OCR text grounding
          • Florence-2 visual grounding (v6, if enabled)
          • History compression with mini model (v6)
          • Self-reflection with mini model (v6)
          • Fact extraction with mini model (v7 fix)
          • UNDO action support (v6)
          • dHash visual-change recovery trigger (v6.1)
          • Stuck detection + auto-recovery
          • HMAC-signed audit log
          • Signal-safe graceful shutdown
        """
        _log("agent", f"Starting task: {task[:120]}")
        db_sid = self.session_db.create_session(task, self.cfg.node_id)
        self.audit.write("task_start", {
            "task":    task,
            "session": db_sid,
            "node":    self.cfg.node_id,
        })

        self.screen.start()

        hb = threading.Thread(
            target=self._heartbeat_loop,
            args=(db_sid,),
            daemon=True,
            name="Nav-Heartbeat",
        )
        hb.start()

        self.memory.prune_expired()

        history:          List[Dict] = self.session_db.load_history(db_sid)
        stuck_recoveries: int        = 0
        result = TaskResult(
            status=TaskStatus.BUDGET,
            message="Task did not complete within step budget",
        )

        try:
            for step in range(1, self.cfg.max_steps + 1):
                if self._shutdown.is_set():
                    result = TaskResult(
                        status=TaskStatus.INTERRUPTED,
                        message="Interrupted by shutdown signal",
                    )
                    break

                _log("agent",
                     f"── Step {step}/{self.cfg.max_steps} "
                     f"[cache hits={self.embed_cache.stats['hits']} "
                     f"misses={self.embed_cache.stats['misses']}]")

                # ── Dynamic frame sampling ────────────────────────────────────
                frames, is_static = self.screen.get_sampled_frames()
                if not frames:
                    _log("agent", "No frames captured — waiting", "warning")
                    time.sleep(1.0)
                    continue

                latest_frame = self.screen.get_latest_frame()

                # ── Stuck detection ───────────────────────────────────────────
                if latest_frame is not None and self.stuck.check(latest_frame):
                    if stuck_recoveries >= self.cfg.max_stuck_recoveries:
                        result = TaskResult(
                            status=TaskStatus.STUCK,
                            message="Agent stuck: exceeded max recovery attempts",
                        )
                        break
                    stuck_recoveries += 1
                    _log("agent",
                         f"Stuck detected — recovery {stuck_recoveries}", "warning")
                    self.memory.add(
                        f"Got stuck on step {step} — try a different approach",
                        importance=1.5,
                    )
                    pyautogui.press("escape")
                    time.sleep(0.5)
                    self.stuck.reset()
                    continue

                # ── Think (compression, reflection, OCR, grounding inside) ────
                try:
                    action, raw, history = self.brain.call(
                        task, frames, history,
                        self.screen.screen_size, step, latest_frame,
                    )
                except (APIError, CircuitOpenError) as exc:
                    _log("agent", f"Brain call failed: {exc}", "error")
                    self.audit.write("error", {"step": step, "error": str(exc)})
                    time.sleep(5.0)
                    continue

                # ── Terminal conditions ───────────────────────────────────────
                if action.type == ActionType.DONE:
                    result = TaskResult(
                        status=TaskStatus.COMPLETED,
                        message=action.message or "Task completed",
                    )
                    self.memory.add(
                        f"Successfully completed: {task[:60]}", importance=2.0
                    )
                    break

                if action.type == ActionType.FAILED:
                    result = TaskResult(
                        status=TaskStatus.FAILED,
                        message=f"Agent failed: {action.message}",
                    )
                    break

                # ── Safety check ──────────────────────────────────────────────
                screen_desc = f"Step {step}, task: {task[:100]}"
                try:
                    self.safety.check(action, screen_desc, self.auto_confirm)
                except SafetyError as exc:
                    _log("agent", f"Safety block: {exc}", "error")
                    self.audit.write("safety_block", {
                        "step":   step,
                        "action": asdict(action),
                        "reason": str(exc),
                    })
                    history.append({"role": "assistant", "content": raw})
                    history.append({
                        "role":    "user",
                        "content": f"[SAFETY BLOCKED]: {exc}",
                    })
                    self.session_db.append_history(
                        db_sid, "safety_block", str(exc), step
                    )
                    continue

                # ── Execute ───────────────────────────────────────────────────
                self.audit.write("action_executed", {
                    "step":   step,
                    "action": asdict(action),
                })

                # v6.1: unpack (success, visual_changed) tuple
                success, visual_changed = self.executor.execute(action)

                if not success:
                    _log("agent",
                         f"Execution failed for action: {action.type}", "warning")

                # ── v6.1: RECOVERY TRIGGER injection ─────────────────────────
                if (
                    success
                    and not visual_changed
                    and action.type in (
                        ActionType.CLICK, ActionType.DOUBLE_CLICK,
                        ActionType.RIGHT_CLICK, ActionType.TYPE,
                        ActionType.HOTKEY, ActionType.SCROLL,
                    )
                ):
                    _log("agent",
                         "Injecting RECOVERY TRIGGER — no visual change detected",
                         "warning")
                    history.append({
                        "role": "user",
                        "content": (
                            "RECOVERY TRIGGER: No visual change detected after "
                            "the last action. The screen is identical to the "
                            "previous frame. The action had ZERO effect. "
                            "Re-plan a completely different strategy or verify "
                            "the element is visible and clickable."
                        ),
                    })

                # ── Update history ────────────────────────────────────────────
                history.append({"role": "assistant", "content": raw})
                self.session_db.append_history(db_sid, "assistant", raw, step)
                self.session_db.update_session(db_sid, step_count=step)

                # Trim history to configured window
                if len(history) > self.cfg.history_limit:
                    history = history[-self.cfg.history_limit:]

                time.sleep(self.cfg.step_pause)

        except Exception as exc:
            result = TaskResult(
                status=TaskStatus.FAILED,
                message=f"Unhandled exception: {exc}",
            )
            _log("agent", f"FATAL: {traceback.format_exc()}", "critical")
            self.audit.write("fatal_error", {
                "error":     str(exc),
                "traceback": traceback.format_exc(),
            })
        finally:
            self._shutdown.set()
            self.screen.stop()
            self.session_db.update_session(
                db_sid, status="done", result=str(result)
            )
            self.audit.write("task_end", {
                "result":      str(result),
                "session":     db_sid,
                "embed_cache": self.embed_cache.stats,
            })
            self.task_queue.push_result({
                "session_id": db_sid,
                "node":       self.cfg.node_id,
                "task":       task,
                "status":     result.status.value,
                "result":     result.message,
            })
            _log("agent", f"Task ended: {result}")

        return result

    def run_worker_mode(self) -> None:
        """
        Distributed worker mode — poll task queue indefinitely.
        Deploy multiple instances across machines; each claims one task at a time.
        """
        _log("agent", f"Worker {self.cfg.node_id} listening on queue…")
        self.screen.start()

        while not self._shutdown.is_set():
            try:
                task_data = self.task_queue.pop_task(timeout=5.0)
                if task_data:
                    task_str = task_data.get("task", "")
                    _log("agent", f"Claimed task from queue: {task_str[:80]}")
                    result = self.run(task_str)
                    _log("agent", f"Worker task result: {result}")
                else:
                    self.task_queue.heartbeat(self.cfg.node_id, "idle")
            except KeyboardInterrupt:
                break
            except Exception as exc:
                _log("agent", f"Worker loop error: {exc}", "error")
                time.sleep(2.0)

        _log("agent", "Worker shutting down")

                   choices=["xai", "voyage", "nomic", "bow"], default=None,

                   help="Disable OCR text grounding")

def main() -> None:
    parser = build_parser()
    args   = parser.parse_args()

    log_level = logging.DEBUG if args.verbose else logging.INFO
    global _logger
    _logger = setup_logging(level=log_level)

    if args.config:
        cfg = Config.from_toml(args.config)
    else:
        cfg = Config.from_env()

    # ── CLI overrides ─────────────────────────────────────────────────────────
    if args.monitor:             cfg.monitor                  = args.monitor
    if args.max_steps:           cfg.max_steps                = args.max_steps
    if args.model:               cfg.model                    = args.model
    if args.safety_model:        cfg.safety_model             = args.safety_model
    if args.memory:              cfg.memory_backend           = args.memory
    if args.redis_url:           cfg.redis_url                = args.redis_url
    if args.embedding_provider:  cfg.embedding_provider       = args.embedding_provider
    if args.no_ocr:              cfg.enable_ocr               = False
    if args.no_reflect:          cfg.reflection_enabled       = False
    if args.no_compress:         cfg.history_compress_enabled = False
    if args.reflection_interval: cfg.reflection_interval      = args.reflection_interval
    if args.embed_cache_size:    cfg.embed_cache_size         = args.embed_cache_size
    if args.grounding:           cfg.grounding_enabled        = True   # v6

    # ── Special modes ─────────────────────────────────────────────────────────
    if args.verify_audit:
        audit = AuditLog(cfg.audit_log_path, cfg.audit_hmac_key)
        valid, invalid = audit.verify_integrity()
        print(f"Audit log: {valid} valid, {invalid} invalid entries")
        sys.exit(0 if invalid == 0 else 1)

    if args.push_task:
        q = DistributedTaskQueue(cfg)
        q.push_task({"task": args.push_task, "submitted_at": time.time()})
        _log("cli", f"Task pushed to queue: {args.push_task}")
        sys.exit(0)

    if not os.environ.get("XAI_API_KEY"):
        sys.exit("ERROR: XAI_API_KEY environment variable must be set")

    agent = NavigatorAgent(
        cfg,
        dry_run=args.dry_run,
        auto_confirm=args.auto_confirm,
    )

    if args.worker:
        agent.run_worker_mode()
    elif args.task:
        result = agent.run(args.task)
        print(f"\nResult: {result}")
        sys.exit(0 if result.succeeded() else 1)
    else:
        parser.print_help()
        sys.exit(1)
